\section{$C$ request/enable, $S$ request/enable}

\begin{hse}
*[[cr->[sr];se-;[~sr];ce-;se+;[~cr];ce+
  \|sr->se-;[~sr];se+
 ]]
\end{hse}

\begin{hse}
*[[cr->c+;[~cr];c-
  \|sr->s+;[~sr];s-
 ]]

*[[c->[sr];ste-;[~sr];ce-;ste+;[~c];ce+
  []s->sfe-;[~s];sfe+
 ]]

*[[~ste|~sfe];se-;[ste&sfe];se+]
\end{hse}

\begin{prs2}
~ce -> ste+
ce & c & sr -> ste-

~s -> sfe+
s -> sfe-
\end{prs2}

\begin{prs2}
ste & ~c -> ce+
~ste & ~sr -> ce-
\end{prs2}

\begin{prs2}
ste & sfe -> se+
~ste | ~sfe -> se-
\end{prs2}

What if we moved the $ste$ after?

*[[cr->c+;[~cr];c-
  \|sr->s+;[~sr];s-
 ]]

*[[c->[sr];ste-;[~sr];ce-;[~c];ste+;ce+
  []s->sfe-;[~s];sfe+
 ]]

*[[~ste|~sfe];se-;[ste&sfe];se+]
\end{hse}
